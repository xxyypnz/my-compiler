{
    set primes;
    set composites;
    set temp;
    int n;
    int i;
    int j;
    bool flag;

    n = 30;
    i = 2;

    while (i <= n) {
        flag = (i in composites);
        if (!flag) {
            write i;
            add primes i;
            j = i + i;
            while (j <= n) {
                add composites j;
                j = j + i;
            }
        }
        i = i + 1;
    }

    write primes;
    write composites;

    temp = primes inter composites;
    flag = isempty(temp);
    write flag;

    flag = (temp == {});
    write flag;

    flag = (primes union composites == primes);
    write flag;
}

{
    set a;
    set b;
    set diff;
    int x;
    bool ok;
    int i;

    a = {2, 4, 6, 8, 10};
    b = {1, 2, 3, 5, 8, 13};
    write a;
    write b;

    diff = {};
    i = 1;
    while (i <= 20) {
        x = 0;
        if (i in a) { x = x + 1; }
        if (i in b) { x = x + 1; }
        if (x == 1) { add diff i; }
        i = i + 1;
    }
    write diff;

    ok = (a inter b == {2, 8});
    write ok;

    ok = (a inter b != {});
    write ok;

    ok = ((a union b) == (b union a));
    write ok;
}

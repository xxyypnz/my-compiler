{
    int x;
    set s;
    bool flag;

    x = 10;
    s = {1, 2};
    flag = true;

    write x;
    write s;
    write flag;

    {
        int x;
        set y;
        x = 20;
        y = {5, 6};

        write x;
        write y;
        write s;
        write flag;

        add y 7;
        write y;

        {
            int x;
            x = 30;
            write x;
            write y;
            add s 10;
            remove s 1;
            write s;
            flag = false;
        }

        write x;
        write y;
        write s;
    }

    write x;
    write s;
    write flag;

    if (flag) {
        write 1;
    }
    else {
        write 0;
    }
}
